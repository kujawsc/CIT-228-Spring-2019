Database Design

Customer: custID(PK), fName, lName, street, zipCode(FK),salesPerson(FK, employeeID), contact(FK), webpage

Contact: contactID(PK),fName, lName, jobTitle, busPhone, mobPhone, faxNum, email

ZipCodes: zipCode(PK), city, state, country

SalePerson: salePersonID(PK), fName, lName

MySQL Commands

Part #1

        CREATE DATABASE company;

Part #2

        USE company;

        CREATE TABLE `contact` (
            `contactID` int(11) INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
            `fName` varchar(50) NULL,
            `lName` varchar(50) NULL,
            `jobTitle` varchar(50) NULL,
            `busPhone` varchar(15) NULL,
            `mobPhone` varchar(15) NULL,
            `faxNum` varchar(25) NULL,
            `email` varchar(50) NULL,
        );
        
        CREATE TABLE `customer` (
            `custID` int(11) INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
            `fName` varchar(50) NULL,
            `lName` varchar(50) NULL,
            `street` varchar(50) NULL,
            `zipCode` int(11) NULL,
            `salesPerson` int(11) NULL,
            `contact` int(11) NULL,
            `webpage` char(50) NULL,
        );
        
        CREATE TABLE ZipCodes(
            zipCode INT(11) NOT NULL PRIMARY KEY,
            city VARCHAR(50),
            state VARCHAR(50),
            country VARCHAR(50)
        );

        CREATE TABLE `saleperson` (
            `salePersonID` int(11) INT NOT NULL PRIMARY KEY AUTO_INCREMENT,
            `fName` varchar(50) DEFAULT NULL,
            `lName` varchar(50) DEFAULT NULL,
        );

        USE company;

        SELECT * 
        FROM customer as c 
        INNER JOIN contact as co
	        ON c.custID = co.contactID
    
        USE company;

        SELECT * 
        FROM customer as c 
        INNER JOIN zipcodes as z
	        ON c.custID = z.zipcode
    
        USE company;

        FROM customer as c 
        INNER JOIN saleperson as s
	        ON c.custID = s.salepersonID
